import React, { useState, useRef, useEffect } from 'react';
import { Brain, Zap, Shield, Skull, TrendingUp, Play, Pause, RotateCcw, Eye, Swords, Trophy, Activity } from 'lucide-react';

const AdversarialArena = () => {
  const canvasRef = useRef(null);
  const [isDrawing, setIsDrawing] = useState(false);
  const [isRunning, setIsRunning] = useState(false);
  const [generation, setGeneration] = useState(0);
  const [battleLog, setBattleLog] = useState([]);
  
  // Arena state
  const [competitors, setCompetitors] = useState([
    { id: 1, name: 'Attention-Alpha', arch: 'ALA', fitness: 95, wins: 0, color: 'bg-blue-500' },
    { id: 2, name: 'Linear-Beta', arch: 'LL', fitness: 82, wins: 0, color: 'bg-green-500' },
    { id: 3, name: 'Residual-Gamma', arch: 'RRL', fitness: 88, wins: 0, color: 'bg-purple-500' },
    { id: 4, name: 'Mixer-Delta', arch: 'MLM', fitness: 91, wins: 0, color: 'bg-orange-500' }
  ]);
  
  const [currentDigit, setCurrentDigit] = useState(null);
  const [predictions, setPredictions] = useState([]);
  const [adversarialMode, setAdversarialMode] = useState(false);
  const [attackStrength, setAttackStrength] = useState(0);
  const [champion, setChampion] = useState(null);
  
  // Drawing functions
  const startDrawing = (e) => {
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    const rect = canvas.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    ctx.beginPath();
    ctx.moveTo(x, y);
    setIsDrawing(true);
  };
  
  const draw = (e) => {
    if (!isDrawing) return;
    
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    const rect = canvas.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    ctx.lineTo(x, y);
    ctx.strokeStyle = adversarialMode ? '#ef4444' : '#ffffff';
    ctx.lineWidth = adversarialMode ? 8 : 12;
    ctx.lineCap = 'round';
    ctx.stroke();
  };
  
  const stopDrawing = () => {
    setIsDrawing(false);
    classifyDrawing();
  };
  
  const clearCanvas = () => {
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    ctx.fillStyle = '#1e293b';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    setPredictions([]);
    setCurrentDigit(null);
  };
  
  // Classification simulation
  const classifyDrawing = () => {
    const digit = Math.floor(Math.random() * 10);
    setCurrentDigit(digit);
    
    // Simulate predictions from each competitor
    const newPredictions = competitors.map(comp => {
      const baseAccuracy = comp.fitness / 100;
      const isCorrect = Math.random() < baseAccuracy;
      const confidence = isCorrect 
        ? 0.7 + Math.random() * 0.3
        : Math.random() * 0.5;
      
      return {
        id: comp.id,
        name: comp.name,
        prediction: isCorrect ? digit : Math.floor(Math.random() * 10),
        confidence: confidence,
        correct: isCorrect
      };
    });
    
    setPredictions(newPredictions);
    
    // Update wins
    setCompetitors(prev => prev.map(comp => {
      const pred = newPredictions.find(p => p.id === comp.id);
      return pred && pred.correct 
        ? { ...comp, wins: comp.wins + 1 }
        : comp;
    }));
    
    addLog(`Digit ${digit} drawn - ${newPredictions.filter(p => p.correct).length}/${competitors.length} models correct`);
  };
  
  // Battle simulation
  const runBattle = () => {
    if (!isRunning) return;
    
    setGeneration(g => g + 1);
    
    // Simulate adversarial attack
    const attackSuccess = Math.random() < (attackStrength / 100);
    
    if (attackSuccess) {
      // Find weakest model
      const weakest = [...competitors].sort((a, b) => a.fitness - b.fitness)[0];
      
      // Evolve replacement
      const newArch = ['ALA', 'RRL', 'MLM', 'LALA', 'ARM'][Math.floor(Math.random() * 5)];
      const newFitness = Math.min(98, weakest.fitness + Math.random() * 10);
      
      setCompetitors(prev => prev.map(comp => 
        comp.id === weakest.id 
          ? { 
              ...comp, 
              name: `${newArch}-${Date.now() % 1000}`,
              arch: newArch,
              fitness: newFitness,
              wins: 0
            }
          : comp
      ));
      
      addLog(`âš”ï¸ ${weakest.name} defeated! Evolved â†’ ${newArch}`, 'attack');
      setAttackStrength(s => Math.max(0, s - 20));
    } else {
      // Models survive, get stronger
      setCompetitors(prev => prev.map(comp => ({
        ...comp,
        fitness: Math.min(99, comp.fitness + Math.random() * 0.5)
      })));
      
      addLog('ðŸ›¡ï¸ All models survived attack round!', 'defend');
      setAttackStrength(s => Math.min(100, s + 5));
    }
    
    // Update champion
    const topModel = [...competitors].sort((a, b) => b.wins - a.wins)[0];
    if (topModel.wins > 5) {
      setChampion(topModel);
    }
  };
  
  const addLog = (message, type = 'info') => {
    setBattleLog(prev => [...prev.slice(-6), { 
      message, 
      type, 
      time: new Date().toLocaleTimeString(),
      gen: generation 
    }]);
  };
  
  useEffect(() => {
    if (isRunning) {
      const interval = setInterval(runBattle, 2000);
      return () => clearInterval(interval);
    }
  }, [isRunning, generation, competitors, attackStrength]);
  
  useEffect(() => {
    clearCanvas();
  }, []);
  
  return (
    <div className="w-full max-w-7xl mx-auto p-6 bg-gradient-to-br from-slate-900 via-purple-900 to-slate-900 rounded-xl shadow-2xl">
      {/* Header */}
      <div className="mb-6 text-center">
        <h1 className="text-4xl font-bold text-white mb-2 flex items-center justify-center gap-3">
          <Swords className="text-red-500" size={36} />
          ADVERSARIAL EVOLUTION ARENA
          <Swords className="text-blue-500" size={36} />
        </h1>
        <p className="text-purple-200">Draw digits â€¢ Watch models battle â€¢ Evolution in real-time</p>
      </div>
      
      {/* Main Arena */}
      <div className="grid grid-cols-3 gap-6 mb-6">
        {/* Drawing Canvas */}
        <div className="col-span-1 bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-purple-500/30">
          <div className="flex items-center justify-between mb-3">
            <h2 className="text-xl font-bold text-white flex items-center gap-2">
              <Eye className="text-cyan-400" size={20} />
              Draw Here
            </h2>
            <button
              onClick={clearCanvas}
              className="px-3 py-1 bg-slate-600 hover:bg-slate-700 text-white rounded text-sm transition-all"
            >
              Clear
            </button>
          </div>
          
          <canvas
            ref={canvasRef}
            width={280}
            height={280}
            className="w-full border-2 border-purple-500/50 rounded cursor-crosshair"
            onMouseDown={startDrawing}
            onMouseMove={draw}
            onMouseUp={stopDrawing}
            onMouseLeave={stopDrawing}
          />
          
          <div className="mt-3 space-y-2">
            <div className="flex items-center gap-2">
              <input
                type="checkbox"
                checked={adversarialMode}
                onChange={(e) => setAdversarialMode(e.target.checked)}
                className="w-4 h-4"
              />
              <label className="text-sm text-purple-200">
                Adversarial Mode (Red Pen)
              </label>
            </div>
            
            {currentDigit !== null && (
              <div className="text-center p-2 bg-purple-500/20 rounded">
                <span className="text-purple-200 text-sm">True Digit:</span>
                <span className="text-white text-2xl font-bold ml-2">{currentDigit}</span>
              </div>
            )}
          </div>
        </div>
        
        {/* Competitors Panel */}
        <div className="col-span-2 bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-purple-500/30">
          <h2 className="text-xl font-bold text-white mb-3 flex items-center gap-2">
            <Brain className="text-purple-400" size={20} />
            Competing Models
          </h2>
          
          <div className="grid grid-cols-2 gap-3">
            {competitors.map(comp => {
              const pred = predictions.find(p => p.id === comp.id);
              return (
                <div 
                  key={comp.id}
                  className={`p-3 rounded-lg border-2 transition-all ${
                    pred?.correct 
                      ? 'border-green-500 bg-green-500/10' 
                      : pred 
                      ? 'border-red-500 bg-red-500/10'
                      : 'border-purple-500/30 bg-slate-700/30'
                  }`}
                >
                  <div className="flex items-center justify-between mb-2">
                    <div className="flex items-center gap-2">
                      <div className={`w-3 h-3 rounded-full ${comp.color}`}></div>
                      <span className="text-white font-semibold text-sm">{comp.name}</span>
                    </div>
                    {champion?.id === comp.id && (
                      <Trophy className="text-yellow-400" size={16} />
                    )}
                  </div>
                  
                  <div className="space-y-1 text-xs">
                    <div className="flex justify-between text-purple-200">
                      <span>Arch:</span>
                      <span className="text-white font-mono">{comp.arch}</span>
                    </div>
                    <div className="flex justify-between text-purple-200">
                      <span>Fitness:</span>
                      <span className="text-white">{comp.fitness.toFixed(1)}%</span>
                    </div>
                    <div className="flex justify-between text-purple-200">
                      <span>Wins:</span>
                      <span className="text-white font-bold">{comp.wins}</span>
                    </div>
                    {pred && (
                      <div className="flex justify-between text-purple-200 pt-1 border-t border-purple-500/30">
                        <span>Predicted:</span>
                        <span className={`font-bold ${pred.correct ? 'text-green-400' : 'text-red-400'}`}>
                          {pred.prediction} ({(pred.confidence * 100).toFixed(0)}%)
                        </span>
                      </div>
                    )}
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      </div>
      
      {/* Control Panel */}
      <div className="grid grid-cols-3 gap-6 mb-6">
        {/* Evolution Controls */}
        <div className="bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-purple-500/30">
          <h3 className="text-lg font-bold text-white mb-3 flex items-center gap-2">
            <Activity className="text-green-400" size={18} />
            Evolution
          </h3>
          
          <div className="space-y-3">
            <button
              onClick={() => setIsRunning(!isRunning)}
              className={`w-full px-4 py-2 rounded-lg font-semibold transition-all flex items-center justify-center gap-2 ${
                isRunning
                  ? 'bg-red-500 hover:bg-red-600 text-white'
                  : 'bg-green-500 hover:bg-green-600 text-white'
              }`}
            >
              {isRunning ? <Pause size={18} /> : <Play size={18} />}
              {isRunning ? 'Pause' : 'Start'} Evolution
            </button>
            
            <div className="text-center p-2 bg-purple-500/20 rounded">
              <div className="text-purple-200 text-sm">Generation</div>
              <div className="text-white text-2xl font-bold">{generation}</div>
            </div>
          </div>
        </div>
        
        {/* Attack Control */}
        <div className="bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-purple-500/30">
          <h3 className="text-lg font-bold text-white mb-3 flex items-center gap-2">
            <Skull className="text-red-400" size={18} />
            Red Team Attack
          </h3>
          
          <div className="space-y-3">
            <div>
              <div className="flex justify-between text-sm text-purple-200 mb-1">
                <span>Strength</span>
                <span className="text-white font-bold">{attackStrength}%</span>
              </div>
              <div className="w-full bg-slate-700 rounded-full h-6 overflow-hidden">
                <div 
                  className="h-full bg-gradient-to-r from-yellow-500 via-orange-500 to-red-500 transition-all duration-500"
                  style={{ width: `${attackStrength}%` }}
                ></div>
              </div>
            </div>
            
            <button
              onClick={() => {
                setAttackStrength(100);
                addLog('ðŸ’€ MANUAL RED TEAM ATTACK INITIATED!', 'attack');
              }}
              className="w-full px-4 py-2 bg-red-600 hover:bg-red-700 text-white rounded-lg font-semibold transition-all flex items-center justify-center gap-2"
            >
              <Zap size={18} />
              Launch Attack
            </button>
          </div>
        </div>
        
        {/* Champion */}
        <div className="bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-purple-500/30">
          <h3 className="text-lg font-bold text-white mb-3 flex items-center gap-2">
            <Trophy className="text-yellow-400" size={18} />
            Current Champion
          </h3>
          
          {champion ? (
            <div className="space-y-2">
              <div className="text-2xl font-bold text-yellow-400">{champion.name}</div>
              <div className="space-y-1 text-sm">
                <div className="flex justify-between text-purple-200">
                  <span>Architecture:</span>
                  <span className="text-white font-mono">{champion.arch}</span>
                </div>
                <div className="flex justify-between text-purple-200">
                  <span>Wins:</span>
                  <span className="text-white font-bold">{champion.wins}</span>
                </div>
                <div className="flex justify-between text-purple-200">
                  <span>Fitness:</span>
                  <span className="text-white">{champion.fitness.toFixed(1)}%</span>
                </div>
              </div>
            </div>
          ) : (
            <div className="text-purple-300 text-center py-8">
              No champion yet<br/>
              <span className="text-sm">(Need 5+ wins)</span>
            </div>
          )}
        </div>
      </div>
      
      {/* Battle Log */}
      <div className="bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-purple-500/30">
        <h3 className="text-lg font-bold text-white mb-3 flex items-center gap-2">
          <TrendingUp className="text-cyan-400" size={18} />
          Battle Log
        </h3>
        
        <div className="space-y-2 max-h-32 overflow-y-auto">
          {battleLog.length === 0 ? (
            <div className="text-purple-300 text-sm italic">Waiting for action...</div>
          ) : (
            battleLog.map((log, idx) => (
              <div 
                key={idx}
                className={`text-sm p-2 rounded flex items-start gap-2 ${
                  log.type === 'attack' ? 'bg-red-500/10 text-red-300' :
                  log.type === 'defend' ? 'bg-green-500/10 text-green-300' :
                  'bg-slate-700/50 text-slate-300'
                }`}
              >
                <span className="text-xs opacity-60 min-w-[60px]">[Gen {log.gen}]</span>
                <span>{log.message}</span>
              </div>
            ))
          )}
        </div>
      </div>
      
      {/* Instructions */}
      <div className="mt-6 p-4 bg-slate-800/30 backdrop-blur rounded-lg border border-purple-500/20">
        <h4 className="text-white font-bold mb-2">How to Play:</h4>
        <div className="text-purple-200 text-sm space-y-1">
          <p>1. <strong className="text-white">Draw digits</strong> on the canvas - watch models compete to classify them</p>
          <p>2. <strong className="text-white">Start Evolution</strong> - models battle in real-time, weak ones get replaced</p>
          <p>3. <strong className="text-white">Launch Attacks</strong> - manually trigger red team to test defenses</p>
          <p>4. <strong className="text-white">Watch the Arena</strong> - see which architectures survive and dominate!</p>
        </div>
      </div>
    </div>
  );
};

export default AdversarialArena;
